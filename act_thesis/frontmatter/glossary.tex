%--- Terms ----------------------------------------------------
\newglossaryentry{superfluid}{
    name        = {superfluidity},
    description = {frictionless flow of a quantum liquid below a critical
                   temperature, characteristic of \helium{} below \SI{2.17}{\kelvin}}
}
\newglossaryentry{nanodroplet}{
    name        = {helium nanodroplet},
    description = {finite cluster of helium atoms, used as a cold, weakly
                   perturbing matrix for embedded molecules}
}
